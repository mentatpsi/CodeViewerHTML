int method2(int[ ] a, int x, int y) { // this method returns integer
    if (x >= y) {
        return a[x];
    }
    else {
        int z = (x + y) / 2;
        return smaller(method2(a, x, z), method2(a, z+1, y));
    }
}

public class Method5 {
    public static int method5(int a, int n) {
        if (n < 1) {
            return 1;
        }
        else {
            int x = n;
            while (x >= 1) {
                a = a + a;
                x = x - 1;
            }
            return a + method5(a, n-1);
        }
    }
}
